or

PalletObjectData
{
	PalletObjectManager Manager
	ObjectData Data
	PutInstance()
	{
		Manager.PutInstance(Data)
	}
}
PalletObjectManager
{
	Binar<> InstanceData
	PutInstance(ObjectData data)
	{
		put data into InstanceData
	}
}
ObjectManager
{
	Binary<PalletObjectManager> Managers
	Binar<ObjectData> Objects
	MakeInstances()
	{
		foreach Objects
		{
			Objects.PutInstance()
		}
	}
}
